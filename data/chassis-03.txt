slides.length - 1, index));
            if (clamped === this.currentSlide) return;

            const target = this.slides[clamped];
            const goingForward = clamped > this.currentSlide;

            if (target.classList.contains('visible')) {
                if (goingForward) {
                    // Forward replay: full reset so the observer can play reveals fresh.
                    target.classList.remove('visible');
                    this._resetSlideReveals(target);
                } else {
                    // Backward rewind: hide any shown steps so the slide
                    // scrolls back in at the start of its stepped content.
                    this._rewindSlide(target);
                }
            }

            this._scrollToSlide(clamped, 'smooth');
        }

        _scrollToSlide(index, behavior) {
            this.viewports[index]?.scrollIntoView({ behavior, block: 'start' });
        }

        _slideIndexFromHash(hash) {
            if (!hash) return -1;
            return this.slides.findIndex((s) => '#' + s.id === hash);
        }

        _syncHash() {
            const slide = this.slides[this.currentSlide];
            if (!slide || !slide.id) return;
            const newHash = '#' + slide.id;
            if (location.hash !== newHash) {
                // replaceState — keeps the URL in sync without polluting history,
                // so the browser back button exits the deck instead of stepping through slides.
                try {
                    history.replaceState(null, '', newHash);
                } catch {
                    // Sandboxed iframes (opaque origin, about:srcdoc) reject URL
                    // mutation with a SecurityError. Hash sync is a nicety — an
                    // unguarded throw here kills the observer callback before it
                    // marks the slide visible and plays its reveals.
                }
            }
        }

        _setupHashNav() {
            // React to external hash changes (anchor link clicks, back/forward if user uses pushState manually)
            window.addEventListener('hashchange', () => {
                const idx = this._slideIndexFromHash(location.hash);
                if (idx >= 0 && idx !== this.currentSlide) this.goTo(idx);
            });
        }

        /* ---------- Custom cursor ----------
           Accent disc that follows the pointer, hidden after idle. */
        _setupCursor() {
            if (!matchMedia('(hover: hover)').matches) return;

            const cursor = document.createElement('div');
            cursor.className = 'chassis-cursor';
            cursor.setAttribute('aria-hidden', 'true');
            document.body.appendChild(cursor);
            document.body.classList.add('chassis-cursor-active');

            const half = 16;  // matches width/height / 2 in CSS
            const IDLE_MS = 2500;
            let idleTimer = null;

            const show = () => {
                cursor.classList.add('visible');
                clearTimeout(idleTimer);
                idleTimer = setTimeout(() => {
                    cursor.classList.remove('visible');
                }, IDLE_MS);
            };
            const hide = () => {
                clearTimeout(idleTimer);
                cursor.classList.remove('visible');
            };

            document.addEventListener('mousemove', (e) => {
                cursor.style.transform = `translate3d(${e.clientX - half}px, ${e.clientY - half}px, 0)`;
                show();
            });
            document.addEventListener('mouseleave', hide);
            document.addEventListener('mouseenter', show);
        }

        /* ---------- Fullscreen ---------- */
        toggleFullscreen() {
            if (!document.fullscreenElement) {
                document.documentElement.requestFullscreen?.().catch(() => {});
            } else {
                document.exitFullscreen?.();
            }
        }

        next() { this.goTo(this.currentSlide + 1); }
        prev() { this.goTo(this.currentSlide - 1); }
        first() { this.goTo(0); }
        last() { this.goTo(this.slides.length - 1); }

        /* ---------- Step-aware advance / retreat ----------
           advance() reveals the next .step on the current slide (if any),
           otherwise advances to the next slide. retreat() hides the last
           revealed step, otherwise goes back to the previous slide. */
        advance() {
            const current = this.slides[this.currentSlide];
            const nextStep = current?.querySelector('.step:not(.step-shown)');
            if (nextStep) {
                nextStep.classList.add('step-shown');
                // No stagger on click-reveal — the user's gesture is the trigger.
                this._playReveal(nextStep, 0);
                return;
            }
            this.next();
        }
        retreat() {
            const current = this.slides[this.currentSlide];
            const shown = current?.querySelectorAll('.step.step-shown');
            if (shown && shown.length > 0) {
                const last = shown[shown.length - 1];
                last.classList.remove('step-shown');
                this._cancelReveal(last);
                this._applyHiddenStyle(last);
                return;
            }
            this.prev();
        }

        /* ---------- Intersection observer ----------
           Fire late (>= 80% in view) so reveals play on a settled slide.
           This is the single source of truth for active-slide state.

           Forward:  add .visible, play reveals. (goTo() has already reset
                     the target to pre-reveal state if this is a re-visit,
                     so the animation is fresh either way.)
           Backward: add .visible, rewind to the start of the stepped
                     content — reveals are shown instantly (no replay) and
                     every step is hidden again so the viewer can step
                     through it fresh. */
        _setupIntersectionObserver() {
            const observer = new IntersectionObserver((entries) => {
                entries.forEach((entry) => {
                    if (!entry.isIntersecting) return;
                    const idx = this.viewports.indexOf(entry.target);
                    if (idx < 0) return;
                    const slide = this.slides[idx];

                    if (idx === this.currentSlide) {
                        // Same-slide re-fire (initial load, resize, fullscreen).
                        if (!slide.classList.contains('visible')) {
                            slide.classList.add('visible');
                        }
                        // Idempotent: only plays if not already marked played.
                        this._playSlideReveals(slide);
                        return;
                    }

                    const goingForward = idx > this.currentSlide;
                    this.currentSlide = idx;
                    this._updateNavDots();
                    this._syncHash();

                    slide.classList.add('visible');
                    if (goingForward) {
                        this._playSlideReveals(slide);
                    } else {
                        this._rewindSlide(slide);
                    }
                });
            }, { threshold: 0.8 });

            this.viewports.forEach((viewport) => observer.observe(viewport));
        }

        /* ---------- Keyboard ---------- */
        _setupKeyboardNav() {
            document.addEventListener('keydown', (e) => {
                // Skip when user is editing text
                if (e.target.isContentEditable || /^(INPUT|TEXTAREA|SELECT)$/.test(e.target.tagName)) return;

                switch (e.key) {
                    // Step-aware: consume steps first, then advance slides
                    case 'ArrowDown':
                    case 'ArrowRight':
                    case ' ':
                        e.preventDefault(); this.advance(); break;
                    case 'ArrowUp':
                    case 'ArrowLeft':
                        e.preventDefault(); this.retreat(); break;
                    // Slide-level: always skip past steps
                    case 'PageDown':
                        e.preventDefault(); this.next(); break;
                    case 'PageUp':
                        e.preventDefault(); this.prev(); break;
                    case 'Home':
                        e.preventDefault(); this.first(); break;
                    case 'End':
                        e.preventDefault(); this.last(); break;
                    case 'f':
                    case 'F':
                        e.preventDefault(); this.toggleFullscreen(); break;
                    case 'r':
                    case 'R':
                        e.preventDefault(); this.first(); break;
                }
            });
        }
