/* =====================================================
   PRESENTATION CHASSIS — slide controller
   Auto-initializes on DOMContentLoaded.
   Exposes window.slides for programmatic control.
   ===================================================== */
(function () {
    'use strict';

    const REVEAL_SELECTOR = '.reveal, .reveal-left, .reveal-right, .reveal-scale, .reveal-blur';
    const STEP_SELECTOR = '.step';
    const ALL_REVEAL_SELECTOR = REVEAL_SELECTOR + ', ' + STEP_SELECTOR;
    const VIEWPORT_CLASS = 'slide-viewport';
    const SCALE_CLASS = 'slide-scale';

    function _detectLang() {
        const param = new URLSearchParams(location.search).get('lang');
        return (param || navigator.language || 'en').slice(0, 2).toLowerCase();
    }

    class SlidePresentation {
        constructor() {
            this.slides = Array.from(document.querySelectorAll('.slide'));
            if (this.slides.length === 0) return;

            // Ensure every slide has an id so hashes + anchor links work.
            // Author-defined ids are preserved; the rest get slide-N (1-indexed).
            this.slides.forEach((slide, idx) => {
                if (!slide.id) slide.id = `slide-${idx + 1}`;
            });
            this.viewports = this._setupViewports();
            this.scales = this.viewports.map((viewport) => viewport.querySelector(`.${SCALE_CLASS}`));

            // Hide every reveal element BEFORE lifting the CSS FOUC guard.
            // After this, .reveals-ready is on <html>, so CSS no longer hides
            // reveals — inline styles keep them hidden until JS animates them.
            this._setupReveals();

            this.navDotsContainer = this._ensureNavDots();
            this.keyboardHint = this._ensureKeyboardHint();
            this.credit = this._ensureCredit();

            this._setupCanvasScale();
            this._setupIntersectionObserver();
            this._setupKeyboardNav();
            this._setupWheelNav();
            this._setupTouchNav();
            this._setupViewportSync();
            this._buildNavDots();
            this._setupHashNav();
            this._setupCursor();
            this._setupPrintSettling();

            // If URL has a matching hash, start there; otherwise start at slide 0.
            const initial = this._slideIndexFromHash(location.hash);
            this.currentSlide = initial >= 0 ? initial : 0;
            if (this.currentSlide !== 0) {
                this._scrollToSlide(this.currentSlide, 'auto');
            }
            // Reveal initial slide immediately (observer may miss it on load).
            const first = this.slides[this.currentSlide];
            first.classList.add('visible');
            this._playSlideReveals(first);
            this._updateNavDots();
        }

        /* ---------- Viewport scaffolding ----------
           Authors write .slide as the 1920×1080 artboard. The chassis wraps
           that authored section in viewport and scale layers. */
        _setupViewports() {
            return this.slides.map((slide) => {
                const existingScale = slide.parentElement?.classList.contains(SCALE_CLASS)
                    ? slide.parentElement
                    : null;
                const existingViewport = existingScale?.parentElement?.classList.contains(VIEWPORT_CLASS)
                    ? existingScale.parentElement
                    : null;
                if (existingViewport) {
                    this._syncViewportBackground(slide, existingViewport);
                    return existingViewport;
                }

                const viewport = document.createElement('div');
                viewport.className = VIEWPORT_CLASS;
                const scale = document.createElement('div');
                scale.className = SCALE_CLASS;

                slide.parentNode.insertBefore(viewport, slide);
                viewport.appendChild(scale);
                scale.appendChild(slide);
                this._syncViewportBackground(slide, viewport);

                return viewport;
            });
        }

        _syncViewportBackground(slide, viewport) {
            const style = getComputedStyle(slide);
            const hasBackgroundImage = style.backgroundImage && style.backgroundImage !== 'none';
            const hasBackgroundColor = !['rgba(0, 0, 0, 0)', 'transparent'].includes(style.backgroundColor);
            if (hasBackgroundImage || hasBackgroundColor) {
                viewport.style.background = style.background;
            } else {
                viewport.style.background = '';
            }
            slide.style.background = 'transparent';
        }

        /* ---------- DOM scaffolding (auto-inserted if missing) ---------- */
        _ensureNavDots() {
            let el = document.getElementById('navDots');
            if (!el) {
                el = document.createElement('nav');
                el.id = 'navDots';
                el.className = 'nav-dots';
                el.setAttribute('aria-label', 'Slide navigation');
                document.body.appendChild(el);
            }
            return el;
        }

        _ensureKeyboardHint() {
            let el = document.querySelector('.keyboard-hint');
            if (!el) {
                const i18n = {
                    en: { navigate: 'navigate', restart: 'restart', fullscreen: 'fullscreen' },
                    es: { navigate: 'navegar',   restart: 'inicio',   fullscreen: 'fullscreen' },
                    fr: { navigate: 'naviguer',  restart: 'début',    fullscreen: 'fullscreen' },
                    de: { navigate: 'blättern',  restart: 'anfang',   fullscreen: 'vollbild' },
                    pt: { navigate: 'navegar',   restart: 'início',   fullscreen: 'fullscreen' },
                    ja: { navigate: '移動',      restart: '先頭',      fullscreen: '全画面' },
                    ko: { navigate: '이동',      restart: '처음',      fullscreen: '전체화면' },
                    zh: { navigate: '翻页',      restart: '重置',      fullscreen: '全屏' },
                    it: { navigate: 'navigare',  restart: 'inizio',   fullscreen: 'fullscreen' },
                    nl: { navigate: 'bladeren',  restart: 'begin',    fullscreen: 'vollscherm' },
                };
                const t = i18n[_detectLang()] || i18n.en;
                el = document.createElement('div');
                el.className = 'keyboard-hint';
                el.innerHTML = `<kbd>←</kbd> <kbd>→</kbd> ${t.navigate} &middot; <kbd>R</kbd> ${t.restart} &middot; <kbd>F</kbd> ${t.fullscreen}`;
                document.body.appendChild(el);
            }
            return el;
        }

        _ensureCredit() {
            let el = document.querySelector('.chassis-credit');
            if (!el) {
                el = document.createElement('div');
                el.className = 'chassis-credit';
                el.setAttribute('aria-hidden', 'true');
                el.innerHTML = '💜 by ClickUp Brain';
                document.body.appendChild(el);
            }
            return el;
        }

        /* ---------- Reveal engine ----------

           transitions. This lets authors put .reveal* / .step on any
           element: animation values live in the animation cascade origin
           and win over normal author declarations, so no wrapper element
           is required to isolate the chassis from author CSS.

           - "hidden":  inline opacity/transform/filter from the 'from'
                        keyframe of the variant. Set on init and on
