------
           ALWAYS preventDefault — letting any native wheel scroll through
           races against scrollIntoView + mandatory snap, and blocks macOS
           two-finger horizontal swipes from triggering browser back/forward.
           Whichever axis has the larger magnitude wins, so vertical scroll
           and horizontal trackpad swipes both drive advance/retreat. Lock
           briefly after a gesture so trackpad inertia (continuous small
           deltas) can't spam next(). */
        _setupWheelNav() {
            let locked = false;
            window.addEventListener('wheel', (e) => {
                if (e.ctrlKey) return;                // preserve browser pinch-zoom (ctrl+wheel)
                e.preventDefault();                   // block native scroll + horizontal back-swipe
                if (locked) return;
                const delta = Math.abs(e.deltaX) > Math.abs(e.deltaY) ? e.deltaX : e.deltaY;
                if (Math.abs(delta) < 20) return;     // ignore tiny flicks
                locked = true;
                if (delta > 0) this.advance(); else this.retreat();
                setTimeout(() => { locked = false; }, 500);
            }, { passive: false });
        }

        /* ---------- Touch / swipe ----------
           Horizontal swipes advance / retreat slides; left = next, right =
           prev (matches stories / carousel conventions). Dominantly
           vertical swipes fall through to native scroll-snap so the
           existing top/bottom behavior keeps working. A time budget
           rejects slow pans that were really aborted scrolls. */
        _setupTouchNav() {
            const MIN_DISTANCE = 50;     // px — below this is a tap or jitter
            const MAX_TIME = 600;        // ms — slower than this is a pan, not a swipe
            const DOMINANCE = 1.2;       // horizontal must beat vertical by this factor

            let startX = 0, startY = 0, startTime = 0;
            let tracking = false;

            window.addEventListener('touchstart', (e) => {
                if (document.body.classList.contains('artifact-editor-editing')) { tracking = false; return; }
                if (e.touches.length !== 1) { tracking = false; return; }
                const t = e.touches[0];
                startX = t.clientX;
                startY = t.clientY;
                startTime = performance.now();
                tracking = true;
            }, { passive: true });

            window.addEventListener('touchend', (e) => {
                if (!tracking) return;
                tracking = false;
                const t = e.changedTouches[0];
                const dx = t.clientX - startX;
                const dy = t.clientY - startY;
                const ax = Math.abs(dx), ay = Math.abs(dy);
                if (ax < MIN_DISTANCE) return;
                if (ax < ay * DOMINANCE) return;
                if (performance.now() - startTime > MAX_TIME) return;
                if (dx < 0) this.advance(); else this.retreat();
            }, { passive: true });

            window.addEventListener('touchcancel', () => { tracking = false; }, { passive: true });
        }

        /* ---------- Canvas scaling ----------
           Each scaler holds a fixed 1920×1080 slide. Compute the
           scale factor that fits it inside the viewport and apply via
           CSS custom property. Runs on init, resize, and fullscreen. */
        _setupCanvasScale() {
            const CANVAS_W = 1920;
            const CANVAS_H = 1080;
            const scales = this.scales;

            const update = () => {
                const scale = Math.min(
                    window.innerWidth / CANVAS_W,
                    window.innerHeight / CANVAS_H
                );
                scales.forEach((el) => {
                    el?.style.setProperty('--slide-scale', scale);
                });
            };

            update();
            window.addEventListener('resize', update);
            document.addEventListener('fullscreenchange', () => {
                requestAnimationFrame(update);
            });
        }

        /* ---------- Viewport sync ----------
           Resize / fullscreenchange → realign current slide. Each slide is
           100vh, so a viewport height change leaves the existing scroll
           position out of alignment with the slide top; scroll-snap doesn't
           re-snap on its own. */
        _setupViewportSync() {
            const realign = () => {
                requestAnimationFrame(() => {
                    this._scrollToSlide(this.currentSlide, 'auto');
                });
            };
            window.addEventListener('resize', realign);
            document.addEventListener('fullscreenchange', realign);
        }

        /* ---------- Nav dots ---------- */
        _buildNavDots() {
            // Clear first — prevents duplicates when DOM is re-serialized/re-opened
            this.navDotsContainer.innerHTML = '';
            this.slides.forEach((slide, idx) => {
                const dot = document.createElement('button');
                dot.setAttribute('aria-label', `Go to slide ${idx + 1}`);
                if (idx === 0) dot.classList.add('active');
                dot.addEventListener('click', () => this.goTo(idx));
                this.navDotsContainer.appendChild(dot);
            });
        }

        _updateNavDots() {
            const dots = this.navDotsContainer.querySelectorAll('button');
            dots.forEach((dot, idx) => {
                dot.classList.toggle('active', idx === this.currentSlide);
            });
        }
    }

    /* ---------- Auto-init ---------- */
    function init() {
        // Each slide canvas is its own editable host for the inline editor.
        document.querySelectorAll('.slide').forEach(function (el) {
            el.classList.add('artifact-editable');
        });
        window.slides = new SlidePresentation();
    }
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', init);
    } else {
        init();
    }
})();

</script>
