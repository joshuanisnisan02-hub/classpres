 re-entry.
           - "played":  animation is running or has run; the class
                        `reveals-played` on the slide makes playback
                        idempotent so observer re-fires (resize,
                        fullscreen) don't restart the animation.
           - "settled": animation canceled, inline styles cleared. Used
                        for backward entries (we skip the animation and
                        show the final state instantly) and for print. */

        _setupReveals() {
            // Lift the CSS FOUC guard first so getComputedStyle reads each
            // element's natural opacity, not the guard's blanket opacity: 0.
            // No paint occurs mid-task — inline hidden styles are applied
            // before the browser yields, so there is no visible flash.
            document.documentElement.classList.add('reveals-ready');
            this.slides.forEach((slide) => {
                slide.querySelectorAll(ALL_REVEAL_SELECTOR).forEach((el) => {
                    el._targetOpacity = parseFloat(getComputedStyle(el).opacity);
                    this._applyHiddenStyle(el);
                });
            });
        }

        _setupPrintSettling() {
            // Browsers snapshot the document for print/PDF export. WAAPI
            // animations don't necessarily complete in that snapshot, so
            // settle every slide to its final state first. After printing,
            // reset — but only for slides that haven't already been viewed,
            // so the audience experience picks up where it left off.
            window.addEventListener('beforeprint', () => {
                this.slides.forEach((slide) => this._settleSlide(slide));
            });
        }

        _readTokens() {
            const cs = getComputedStyle(document.documentElement);
            const rawDuration = parseFloat(cs.getPropertyValue('--reveal-duration'));
            const duration = Number.isFinite(rawDuration) ? rawDuration * 1000 : 800;
            const ease = cs.getPropertyValue('--reveal-ease').trim() || 'cubic-bezier(0.16, 1, 0.3, 1)';
            const distance = cs.getPropertyValue('--reveal-distance').trim() || '28px';
            const reducedMotion = matchMedia('(prefers-reduced-motion: reduce)').matches;
            return {
                duration: reducedMotion ? 1 : duration,
                ease,
                distance,
            };
        }

        _getRevealDelay(el) {
            // Use getAttribute('class') rather than el.className: on SVG
            // elements, `.className` is an SVGAnimatedString (not a string)
            // and calling .match on it throws. getAttribute is uniform
            // across HTML and SVG.
            const classes = el.getAttribute('class') || '';
            const m = classes.match(/\bd([1-8])\b/);
            if (!m) return 0;
            // Matches CSS .dN: d1 = 100ms, d2 = 250ms, +150ms per step.
            return 100 + (parseInt(m[1], 10) - 1) * 150;
        }

        _getRevealKeyframes(el, distance) {
            const to = el._targetOpacity ?? 1;
            if (el.classList.contains('reveal-left')) {
                return [
                    { opacity: 0, transform: `translateX(calc(-1 * ${distance}))` },
                    { opacity: to, transform: 'translateX(0)' },
                ];
            }
            if (el.classList.contains('reveal-right')) {
                return [
                    { opacity: 0, transform: `translateX(${distance})` },
                    { opacity: to, transform: 'translateX(0)' },
                ];
            }
            if (el.classList.contains('reveal-scale')) {
                return [
                    { opacity: 0, transform: 'scale(0.88)' },
                    { opacity: to, transform: 'scale(1)' },
                ];
            }
            if (el.classList.contains('reveal-blur')) {
                return [
                    { opacity: 0, filter: 'blur(10px)' },
                    { opacity: to, filter: 'blur(0)' },
                ];
            }
            // Default: .reveal or .step (slide up + fade)
            return [
                { opacity: 0, transform: `translateY(${distance})` },
                { opacity: to, transform: 'translateY(0)' },
            ];
        }

        _applyHiddenStyle(el) {
            const { distance } = this._readTokens();
            const from = this._getRevealKeyframes(el, distance)[0];
            if ('opacity' in from) el.style.opacity = from.opacity;
            if ('transform' in from) el.style.transform = from.transform;
            if ('filter' in from) el.style.filter = from.filter;
        }

        _clearHiddenStyle(el) {
            el.style.opacity = '';
            el.style.transform = '';
            el.style.filter = '';
        }

        _cancelReveal(el) {
            if (el._revealAnim) {
                el._revealAnim.cancel();
                el._revealAnim = null;
            }
        }

        _playReveal(el, delayOverride) {
            this._cancelReveal(el);
            const { duration, ease, distance } = this._readTokens();
            const delay = typeof delayOverride === 'number' ? delayOverride : this._getRevealDelay(el);
            const keyframes = this._getRevealKeyframes(el, distance);
            // fill: 'backwards' applies the 'from' keyframe during the
            // stagger delay so the element stays hidden until its turn,
            // without pinning the 'to' keyframe after the animation ends.
            // This lets co-existing author CSS (e.g. a `.pulse` CSS
            // animation on the same element) take over naturally once the
            // entrance completes — otherwise our fill would mask anything
            // the author wanted to happen post-reveal.
            const anim = el.animate(keyframes, {
                duration,
                delay,
                easing: ease,
                fill: 'backwards',
            });
            el._revealAnim = anim;
            // Safe to clear inline 'from' styles now — the animation's
            // backwards-fill holds the element at the 'from' keyframe for
            // the duration of any stagger delay.
            this._clearHiddenStyle(el);
        }

        _playSlideReveals(slide) {
            if (slide.classList.contains('reveals-played')) return;
            slide.classList.add('reveals-played');
            slide.querySelectorAll(REVEAL_SELECTOR).forEach((el) => this._playReveal(el));
            // .step elements stay hidden until the user advances.
        }

        _resetSlideReveals(slide) {
            // Forward re-entry: cancel in-flight animations, re-hide
            // everything, and reset step state so the slide replays cleanly
            // when the observer next adds .visible.
            slide.classList.remove('reveals-played');
            slide.querySelectorAll(ALL_REVEAL_SELECTOR).forEach((el) => {
                this._cancelReveal(el);
                this._applyHiddenStyle(el);
            });
            slide.querySelectorAll('.step.step-shown').forEach((el) => el.classList.remove('step-shown'));
        }

        _settleSlide(slide) {
            // Print / PDF export: jump to the final state with everything
            // shown, so every step appears in the exported document.
            slide.classList.add('reveals-played');
            slide.querySelectorAll(ALL_REVEAL_SELECTOR).forEach((el) => {
                this._cancelReveal(el);
                this._clearHiddenStyle(el);
            });
            slide.querySelectorAll('.step').forEach((el) => el.classList.add('step-shown'));
        }

        _rewindSlide(slide) {
            // Backward re-entry: show reveals instantly (no replay — the
            // viewer already saw the intro) but hide every step, so the
            // slide is positioned at the start of its stepped content and
            // can be stepped through again.
            slide.classList.add('reveals-played');
            slide.querySelectorAll(REVEAL_SELECTOR).forEach((el) => {
                this._cancelReveal(el);
                this._clearHiddenStyle(el);
            });
            slide.querySelectorAll('.step').forEach((el) => {
                el.classList.remove('step-shown');
                this._cancelReveal(el);
                this._applyHiddenStyle(el);
            });
        }

        /* ---------- Navigation ----------
           goTo scrolls to the target. Both forward-replay and backward-
           rewind need to mutate the target BEFORE scrolling — while it's
           still offscreen — so the viewer never sees the old state flash
           into view. Resetting after the slide is in view causes a visible
           "show then hide" blink; the IntersectionObserver fires too late
           (80% threshold) to catch it. */
